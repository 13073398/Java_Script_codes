/*
  DOM [CSS]
  - style
  - cssText
  - removeProperty(propertyName) [Inline, Stylesheet]
  - setProperty(propertyName, value, priority)
*/



let mohamed=document.getElementById("my-div");
// to set  css style inline style
mohamed.style.fontSize="100px";
// to set  css style inline style
mohamed.style.color="red"
// to set  css style in css file
mohamed.style.cssText="font-weight:bold ; color:green; font-size:100px"
// to set  css style inline style By  Method
mohamed.style.removeProperty("color")
// to set  css style inline style By  Method
mohamed.style.setProperty("font-size","20px" ,"important")
 // to remove style from in line style
document.styleSheets[0].cssRules[0].style.removeProperty("line-height")
 // to add style from in line style
document.styleSheets[0].cssRules[0].style.setProperty("background-color", "red")
