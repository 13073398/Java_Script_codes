let mohamed=document.querySelector("div");

console.log(mohamed.childNodes)

console.log(mohamed.childNodes[0])

console.log(mohamed.firstChild)
console.log(mohamed.lastChild)
console.log(mohamed.firstElementChild)
console.log(mohamed.lastElementChild)
